&emsp;&emsp;

``` c++
typedef void  (*Init_board)(void);
void Init_function(void);
const Init_board Init_Sequnce[] = 
{
  Init_function,
  NULL,
};
int Funs_Init(void)
{
  const Init_board *call = Init_Sequnce;

  for(; *call; ++call)
    (*call)();

  return 0;
}
```
